function(add_edl_library name)
  # Trusted (enclave-side) codegen + library
  add_custom_command(
    OUTPUT
      ${CMAKE_CURRENT_BINARY_DIR}/${name}_t.c
      ${CMAKE_CURRENT_BINARY_DIR}/${name}_t.h
      ${CMAKE_CURRENT_BINARY_DIR}/${name}_args.h
    DEPENDS
      ${CMAKE_CURRENT_SOURCE_DIR}/${name}.edl
    COMMAND
      openenclave::oeedger8r --trusted ${CMAKE_CURRENT_SOURCE_DIR}/${name}.edl
        --trusted-dir ${CMAKE_CURRENT_BINARY_DIR}
        --search-path ${OE_INCLUDEDIR}
        --search-path ${OE_INCLUDEDIR}/openenclave/edl/sgx
  )
  add_library(
    edl_${name}_trusted
    OBJECT
    ${CMAKE_CURRENT_BINARY_DIR}/${name}_t.c
  )
  target_include_directories(
    edl_${name}_trusted
    PUBLIC
      ${CMAKE_CURRENT_BINARY_DIR}
  )
  target_link_libraries(
    edl_${name}_trusted
    PUBLIC
      project_includes
      openenclave::oeenclave
      openenclave::oecryptombedtls
  )

  # Untrusted (host-side) codegen + library
  add_custom_command(
    OUTPUT
      ${CMAKE_CURRENT_BINARY_DIR}/${name}_u.c
      ${CMAKE_CURRENT_BINARY_DIR}/${name}_u.h
    DEPENDS
      ${CMAKE_CURRENT_BINARY_DIR}/${name}_args.h   # generated by trusted step above
      ${CMAKE_CURRENT_SOURCE_DIR}/${name}.edl
    COMMAND
      openenclave::oeedger8r --untrusted ${CMAKE_CURRENT_SOURCE_DIR}/${name}.edl
        --untrusted-dir ${CMAKE_CURRENT_BINARY_DIR}
        --search-path ${OE_INCLUDEDIR}
        --search-path ${OE_INCLUDEDIR}/openenclave/edl/sgx
  )
  add_library(
    edl_${name}_untrusted
    OBJECT
    ${CMAKE_CURRENT_BINARY_DIR}/${name}_u.c
  )
  target_include_directories(
    edl_${name}_untrusted
    PUBLIC
      ${CMAKE_CURRENT_BINARY_DIR}
  )
  target_link_libraries(
    edl_${name}_untrusted
    PUBLIC
      project_includes
      openenclave::oehost
  )
endfunction()

function(sign_enclave name)
  cmake_parse_arguments(arg "" "KEY_TARGET" "" ${ARGN})

  # Generate a private key for signing if needed
  if(arg_KEY_TARGET)
    get_target_property(key_file ${arg_KEY_TARGET} SIGNING_KEY)
    set(key_dep)
  else()
    set(key_file ${CMAKE_CURRENT_BINARY_DIR}/${name}_private.pem)
    add_custom_command(
      OUTPUT
        ${key_file}
      COMMAND
        openssl genrsa -out ${key_file} -3 3072
    )
    set(key_dep ${key_file})
  endif()


  # Sign the enclave with the generated key and config file
  add_custom_command(
    OUTPUT
      ${CMAKE_CURRENT_BINARY_DIR}/${name}.signed
    DEPENDS
      ${name}
      ${CMAKE_CURRENT_SOURCE_DIR}/sgx.conf
      ${key_dep}
    COMMAND
      openenclave::oesign sign
        --enclave-image $<TARGET_FILE:${name}>
        --config-file ${CMAKE_CURRENT_SOURCE_DIR}/sgx.conf
        --key-file ${key_file}
  )

  # Drive the signing as part of the default build
  add_custom_target(
    ${name}_signed
    ALL
    DEPENDS
      ${CMAKE_CURRENT_BINARY_DIR}/${name}.signed
  )
  if(arg_KEY_TARGET)
    add_dependencies(${name}_signed ${arg_KEY_TARGET})
  endif()
endfunction()


add_edl_library(hello)
add_subdirectory(hello)
add_edl_library(datasealing)
add_subdirectory(datasealing)
